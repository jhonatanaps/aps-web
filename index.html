function showTab(tabId) { let tabs = document.querySelectorAll('.tab-content'); tabs.forEach(tab => tab.style.display = 'none'); document.getElementById(tabId).style.display = 'block'; } function gerarNumeros() { let milhar = document.getElementById('milhar').value; let centena = document.getElementById('centena').value; let milharGerada = milhar ? milhar + Math.floor(Math.random() * 1000) : 'Sem entrada'; let centenaGerada = centena ? centena + Math.floor(Math.random() * 100) : 'Sem entrada'; document.getElementById('milhares').innerText = milharGerada; document.getElementById('centenas').innerText = centenaGerada; } document.addEventListener("DOMContentLoaded", () => { const bichos = ["Avestruz", "Águia", "Burro", "Borboleta", "Cachorro", "Cabra", "Carneiro", "Camelo", "Cobra", "Coelho", "Cavalo", "Elefante", "Galo", "Gato", "Jacaré", "Leão", "Macaco", "Porco", "Pavão", "Peru", "Touro", "Tigre", "Urso", "Veado", "Vaca"]; let bichoButtons = document.getElementById("bicho-buttons"); let contadorLista = document.getElementById("contador-lista"); let contadores = {}; bichos.forEach(bicho => { let btn = document.createElement("button"); btn.innerText = bicho; btn.onclick = () => { contadores[bicho] = (contadores[bicho] || 0) + 1; atualizarContadores(); }; bichoButtons.appendChild(btn); }); function atualizarContadores() { contadorLista.innerHTML = ""; for (let bicho in contadores) { let item = document.createElement("li"); item.innerText = `${bicho}: ${contadores[bicho]}`; contadorLista.appendChild(item); } } });
